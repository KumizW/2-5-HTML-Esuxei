#include <stdio.h>
int checkPrime(int too) {
    for (int i=2;i<too;i++) {
        if (too%i==0)
            return 0;
    }
    return 1;
}
int main() {
    int too;
    printf("too oruul:");scanf("%d",&too);
    int result=checkPrime(too);
    if(result==1)
    printf("anxnii too");
    else if(result==0)
    printf("anxnii too bish");
    return 0;
}
