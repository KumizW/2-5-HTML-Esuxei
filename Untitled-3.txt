#include <stdio.h>
#include <string.h>
struct Student {
    int id;
    char name[50];
    int age;
    float gpa;
};
int main() {
    int n, i;
    printf("Enter number of students: ");
    scanf("%d", &n);
    struct Student s[n];
    for(i=0;i<n;i++) 
    {
        printf("\n--- Enter data for student %d ---\n", i+1);
        printf("ID: ");
        scanf("%d", &s[i].id);
        printf("Name: ");
        scanf(" %[^\n]", s[i].name);
            printf("Age: ");
        scanf("%d",&s[i].age);
        printf("GPA: ");
        scanf("%f",&s[i].gpa);
    }
    printf("\nAll Students:\n");
    printf("ID\tName\t\tAge\tGPA\n");
    for(i=0;i<n;i++) {
        printf("%d\t%-10s\t%d\t%.2f\n", s[i].id, s[i].name, s[i].age, s[i].gpa);
    }
    float maxGPA=s[0].gpa;
    for(i=1;i<n;i++) {
        if(s[i].gpa>maxGPA)
            maxGPA=s[i].gpa;
    }
    printf("\nStudent(s) with highest GPA:\n");
    for(i=0;i<n;i++) {
        if(s[i].gpa==maxGPA) {
            printf("ID: %d, Name: %s, GPA: %.2f\n", s[i].id, s[i].name, s[i].gpa);
        }
    }
    return 0;
}
